使用 Qexo 编辑博客的时候，图片都是通过网络链接引用，一般存储到阿里云oss中，如果oss到期了，要把md文件中的图片转为本地图片，那么如何操作呢 ？

可通过如下代码完成：

package main

import (
	"bufio"
	"flag"
	"fmt"
	"io"
	"net/http"
	"os"
	"path/filepath"
	"regexp"
	"strings"
	"time"
)

// 匹配 Markdown 中的网络图片链接，捕获 alt 文本和 URL
var mdImgRegex = regexp.MustCompile(`!\[(.*?)\]\((https?://.*?)\)`)

func printWithLine(msg string) {
	line := strings.Repeat("-", len(msg))
	fmt.Println(line)
	fmt.Println(msg)
}

// 下载图片，按 image+时间戳+后缀命名，返回保存路径和文件名（不含路径）
func downloadImage(url, saveDir string) (string, error) {

	urlNoParam := url
	if idx := strings.Index(url, "?"); idx != -1 {
		urlNoParam = url[:idx]
	}
	ext := filepath.Ext(urlNoParam)
	if ext == "" {
		ext = ".jpg"
	}

	timestamp := strings.ReplaceAll(strings.ReplaceAll(time.Now().Format("20060102150405.0000"), ".", ""), "-", "")
	fileName := fmt.Sprintf("image-%s%s", timestamp, ext)
	savePath := filepath.Join(saveDir, fileName)

	if _, err := os.Stat(saveDir); os.IsNotExist(err) {
		if err := os.MkdirAll(saveDir, os.ModePerm); err != nil {
			return "", fmt.Errorf("创建目录失败: %w", err)
		}
	}

	resp, err := http.Get(url)
	if err != nil {
		return "", fmt.Errorf("下载失败: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return "", fmt.Errorf("HTTP状态错误: %s", resp.Status)
	}

	out, err := os.Create(savePath)
	if err != nil {
		return "", fmt.Errorf("创建文件失败: %w", err)
	}
	defer out.Close()

	_, err = io.Copy(out, resp.Body)
	if err != nil {
		return "", fmt.Errorf("保存图片失败: %w", err)
	}

	// 为了避免时间戳重复，休眠10毫秒
	time.Sleep(10 * time.Millisecond)

	return savePath, nil
}

// 处理 Markdown 文件中的网络图片，替换 [] 和 () 内容
func processMdPic(mdPath string) {
	msg := fmt.Sprintf("正在处理文件 %s", mdPath)
	printWithLine(msg)

	mdDir := filepath.Dir(mdPath)
	mdName := strings.TrimSuffix(filepath.Base(mdPath), filepath.Ext(mdPath))
	targetImageFolder := filepath.Join(mdDir, mdName)

	file, err := os.Open(mdPath)
	if err != nil {
		fmt.Println("打开文件失败:", err)
		return
	}
	defer file.Close()

	scanner := bufio.NewScanner(file)
	findPic := false
	var updatedLines []string
	count := 1

	for scanner.Scan() {
		line := scanner.Text()
		newLine := line

		matches := mdImgRegex.FindAllStringSubmatch(line, -1)
		for _, match := range matches {
			if len(match) >= 3 {
				findPic = true
				imageURL := match[2]
				fmt.Printf("提取到网络图片链接[%d]: %s\n", count, imageURL)

				savedPath, err := downloadImage(imageURL, targetImageFolder)
				if err != nil {
					fmt.Println("下载失败:", err)
					continue
				}

				imageFileName := filepath.Base(savedPath)
				fileNameNoExt := strings.TrimSuffix(imageFileName, filepath.Ext(imageFileName))

				relativePath := fmt.Sprintf("./%s/%s", mdName, imageFileName)

				// 构造新的 Markdown 图片语法
				newContent := fmt.Sprintf("![%s](%s)", fileNameNoExt, relativePath)

				oldContent := match[0]

				newLine = strings.Replace(newLine, oldContent, newContent, 1)

				fmt.Printf("下载成功，保存为: %s\n", savedPath)
				fmt.Printf("替换 Markdown 内容为: %s\n", newContent)
				count++
			}
		}
		updatedLines = append(updatedLines, newLine)
	}

	if !findPic {
		fmt.Println("Notice：未提取到网络图片链接")
	}

	if err := scanner.Err(); err != nil {
		fmt.Println("读取文件出错:", err)
		return
	}

	err = os.WriteFile(mdPath, []byte(strings.Join(updatedLines, "\n")), 0644)
	if err != nil {
		fmt.Println("写入 Markdown 文件失败:", err)
		return
	}

	if !findPic {
		fmt.Println("⚠️ 无需对文件进行处理")
	} else {
		fmt.Println("✅ 文件处理完成并成功替换网络图片链接")
	}
	fmt.Println()
}

func getFolder(folderPath string) {
	if _, err := os.Stat(folderPath); os.IsNotExist(err) {
		fmt.Println("-p 指定的目录不存在:", folderPath)
		return
	}

	entries, err := os.ReadDir(folderPath)
	if err != nil {
		fmt.Println("读取目录失败:", err)
		return
	}

	for _, entry := range entries {
		if entry.Type().IsRegular() && strings.HasSuffix(entry.Name(), ".md") {
			fullPath := filepath.Join(folderPath, entry.Name())
			processMdPic(fullPath)
		}
	}
}

func backupFolder(src, dst string) error {
	return filepath.Walk(src, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return err
		}

		relPath, err := filepath.Rel(src, path)
		if err != nil {
			return err
		}

		destPath := filepath.Join(dst, relPath)

		if info.IsDir() {
			return os.MkdirAll(destPath, os.ModePerm)
		}

		srcFile, err := os.Open(path)
		if err != nil {
			return err
		}
		defer srcFile.Close()

		destFile, err := os.Create(destPath)
		if err != nil {
			return err
		}
		defer destFile.Close()

		_, err = io.Copy(destFile, srcFile)
		return err
	})
}

func main() {
	path := flag.String("p", "", "指定需要处理的文件夹路径")
	flag.Parse()

	if *path == "" {
		fmt.Println("请使用 -p 参数指定路径，例如：test.exe -p \"D:\\\\Blog\\\\content\\\\post\"")
		os.Exit(1)
	}

	timestamp := time.Now().Format("20060102-150405")
	backupDst := filepath.Join("D:\\", "post-backup-"+timestamp)
	fmt.Printf("⏳ 正在备份原始文件夹 %s 到 %s...\n", *path, backupDst)
	err := backupFolder(*path, backupDst)
	if err != nil {
		fmt.Println("❌ 自动备份失败:", err)
		os.Exit(1)
	}
	fmt.Println("✅ 备份完成。")

	fmt.Println("执行前务必先完成对原博客的备份操作 (已完成备份/N) :")

	reader := bufio.NewReader(os.Stdin)
	input, err := reader.ReadString('\n')
	if err != nil {
		fmt.Println("读取输入失败:", err)
		os.Exit(1)
	}

	input = strings.TrimSpace(input)
	if strings.ToUpper(input) != "已完成备份" {
		fmt.Println("已取消操作。")
		os.Exit(0)
	}

	getFolder(*path)
}
